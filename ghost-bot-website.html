index.js 
const { default: makeWASocket, useMultiFileAuthState, DisconnectReason, fetchLatestBaileysVersion } = require('@whiskeysockets/baileys');
const { Boom } = require('@hapi/boom');
const pino = require('pino');
const path = require('path');
const fs = require('fs');
require('dotenv').config();

// Load commands
const menuCmd = require('./commands/menu');
const downloadCmd = require('./commands/download');
const mediaCmd = require('./commands/media');
const groupCmd = require('./commands/group');

const PREFIX = process.env.PREFIX || '.';
const OWNER = process.env.OWNER_NUMBER;
const BOT_NAME = process.env.BOT_NAME || 'GHOST';
const SESSION_DIR = './session';

async function startBot() {
  const { state, saveCreds } = await useMultiFileAuthState(SESSION_DIR);
  const { version } = await fetchLatestBaileysVersion();

  const sock = makeWASocket({
    version,
    logger: pino({ level: 'silent' }),
    auth: state,
    printQRInTerminal: false,
    browser: [`${BOT_NAME} BOT`, 'Chrome', '3.0'],
    generateHighQualityLinkPreview: true,
  });

  // Connection update handler
  sock.ev.on('connection.update', async (update) => {
    const { connection, lastDisconnect, qr } = update;
    if (connection === 'close') {
      const shouldReconnect = (new Boom(lastDisconnect?.error))?.output?.statusCode !== DisconnectReason.loggedOut;
      if (shouldReconnect) startBot();
    } else if (connection === 'open') {
      console.log(`\n👻 ${BOT_NAME} BOT Online!\n`);
    }
  });

  sock.ev.on('creds.update', saveCreds);

  // Message handler
  sock.ev.on('messages.upsert', async ({ messages }) => {
    for (const msg of messages) {
      if (!msg.message || msg.key.fromMe) continue;
      const text = msg.message?.conversation || msg.message?.extendedTextMessage?.text || '';
      if (!text.startsWith(PREFIX)) continue;
      const [cmd, ...args] = text.slice(PREFIX.length).trim().split(' ');
      const command = cmd.toLowerCase();
      const ctx = { sock, msg, args, text, OWNER, BOT_NAME, PREFIX };
      try {
        if (['menu','help','ping','alive','owner'].includes(command)) await menuCmd(command, ctx);
        else if (['song','video','tiktok','insta','fb'].includes(command)) await downloadCmd(command, ctx);
        else if (['sticker','toimg','ai','weather','lyrics'].includes(command)) await mediaCmd(command, ctx);
        else if (['kick','promote','demote','tagall'].includes(command)) await groupCmd(command, ctx);
      } catch(err) {
        console.error(err);
        await sock.sendMessage(msg.key.remoteJid, { text: '❌ Error: ' + err.message });
      }
    }
  });
}

startBot();

const os = require('os');

module.exports = async (cmd, { sock, msg, BOT_NAME, OWNER, PREFIX }) => {
  const jid = msg.key.remoteJid;
  const reply = text => sock.sendMessage(jid, { text }, { quoted: msg });

  switch(cmd) {
    case 'menu':
      await reply(`
╔══════════════════╗
║   👻 *${BOT_NAME} BOT*   ║
╚══════════════════╝

⚡ *GENERAL*
${PREFIX}ping — Bot speed
${PREFIX}alive — Online status
${PREFIX}owner — Owner info
${PREFIX}runtime — Bot uptime

🎵 *DOWNLOADS*
${PREFIX}song [name] — MP3
${PREFIX}video [url] — YouTube
${PREFIX}tiktok [url] — TikTok
${PREFIX}insta [url] — Instagram
${PREFIX}fb [url] — Facebook

🎨 *MEDIA*
${PREFIX}sticker — Img to sticker
${PREFIX}toimg — Sticker to img
${PREFIX}ai [text] — AI chat
${PREFIX}weather [city] — Weather
${PREFIX}lyrics [song] — Lyrics

👑 *GROUP (Admin)*
${PREFIX}kick @user
${PREFIX}promote @user
${PREFIX}tagall
${PREFIX}antilink on/off

╔══════════════════╗
║  Bot by GHOST ™  ║
╚══════════════════╝`);
      break;

    case 'ping':
      const start = Date.now();
      await sock.sendMessage(jid, { react: { text: '⚡', key: msg.key } });
      await reply(`🏓 Pong! ${Date.now() - start}ms`);
      break;

    case 'alive':
      await reply(`👻 *${BOT_NAME} BOT* Online!\n⚡ RAM: ${(process.memoryUsage().heapUsed/1024/1024).toFixed(2)}MB\n🖥️ OS: ${os.platform()}`);
      break;

    case 'owner':
      await reply(`👑 Bot Owner\n📞 wa.me/${OWNER}`);
      break;
  }
};
commands/download.js
const ytdl = require('ytdl-core');
const ytSearch = require('yt-search');
const axios = require('axios');
const fs = require('fs');
const path = require('path');

module.exports = async (cmd, { sock, msg, args }) => {
  const jid = msg.key.remoteJid;
  const reply = t => sock.sendMessage(jid, { text: t }, { quoted: msg });
  const query = args.join(' ');

  if (cmd === 'song') {
    if (!query) return reply('🎵 Song name denn!\nExample: .song Shape of You');
    await reply('🔍 Searching...');
    try {
      const results = await ytSearch(query);
      const video = results.videos[0];
      if (!video) return reply('❌ Song nathuwa');
      await reply(`🎵 Found: *${video.title}*\n⏱️ ${video.timestamp}\n⬇️ Downloading...`);
      const stream = ytdl(video.url, { filter: 'audioonly', quality: 'highestaudio' });
      const tmpFile = path.join('/tmp', `song_${Date.now()}.mp3`);
      const writer = fs.createWriteStream(tmpFile);
      stream.pipe(writer);
      await new Promise((res, rej) => { writer.on('finish', res); writer.on('error', rej); });
      await sock.sendMessage(jid, {
        audio: fs.readFileSync(tmpFile),
        mimetype: 'audio/mpeg',
        fileName: `${video.title}.mp3`,
        ptt: false
      }, { quoted: msg });
      fs.unlinkSync(tmpFile);
    } catch(e) { await reply('❌ Download failed: ' + e.message); }
  }

  else if (cmd === 'video') {
    if (!query) return reply('📹 URL denn!\nExample: .video https://youtube.com/...');
    await reply('⬇️ Downloading video...');
    try {
      const info = await ytdl.getInfo(query);
      const stream = ytdl(query, { filter: 'videoandaudio', quality: '18' });
      const tmpFile = path.join('/tmp', `vid_${Date.now()}.mp4`);
      const writer = fs.createWriteStream(tmpFile);
      stream.pipe(writer);
      await new Promise((res,rej) => { writer.on('finish',res); writer.on('error',rej); });
      await sock.sendMessage(jid, {
        video: fs.readFileSync(tmpFile),
        caption: `🎬 ${info.videoDetails.title}`,
        fileName: `${info.videoDetails.title}.mp4`
      }, { quoted: msg });
      fs.unlinkSync(tmpFile);
    } catch(e) { await reply('❌ Download failed: ' + e.message); }
  }

  else if (cmd === 'tiktok') {
    if (!query) return reply('📱 TikTok URL denn!');
    await reply('⬇️ TikTok downloading...');
    try {
      const { data } = await axios.get(`https://www.tikwm.com/api/?url=${encodeURIComponent(query)}`);
      if (!data.data?.play) return reply('❌ TikTok download nathuwa');
      const res = await axios.get(data.data.play, { responseType: 'arraybuffer' });
      await sock.sendMessage(jid, {
        video: Buffer.from(res.data),
        caption: `🎵 ${data.data.title || 'TikTok Video'}`
      }, { quoted: msg });
    } catch(e) { await reply('❌ Error: ' + e.message); }
  }
};
